#define UPDATE_TIME 20

int servoPin = 5;

void setup() {
    pinMode(5, OUTPUT);
    Serial.begin(9600);
}

void loop() {
//  analogWrite(5, HIGH);
//  delay (2000);
//  analogWrite(5, LOW);
//  delay (2000);

  for(int i=0;i<180;i++){
    analogWrite(servoPin,convertirAngleEnPWM(i));
    delay(20);
  }
delay(200);
  for(int i=180;i>=0;i--){
    analogWrite(servoPin,convertirAngleEnPWM(i));
    delay(20);
  }
delay(200);
exit(0);
}


int convertirAngleEnPWM(int ang){
  float a = 255.0/180.0;
  return (a*ang);
}
